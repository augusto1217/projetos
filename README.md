#### Instalar Node e Angular com NVM no Linux:

> wget -qO- https://raw.githubusercontent.com/nvm-sh/nvm/v0.39.5/install.sh | bash
source ~/.bashrc
nvm --version
nvm install --lts
nvm use --lts
npm install --global @angular/cli
